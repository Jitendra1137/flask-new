pipeline {
    agent { label 'node-2' }

    stages {
        stage('Code') {
            steps {
                echo 'Fetching code...'
                git url:"https://github.com/Jitendra1137/flask-new.git", branch:"main"
                echo "code cloning successfully"
            }
        }

        stage('Cleanup') {
            steps {
                echo 'Cleaning previous deployments...'
                sh """
                    docker-compose down || true
                    docker stop simple-flask-app || true
                    docker rm simple-flask-app || true
                """
            }
        }

        stage('Build') {
            steps {
                echo 'Building the application...'
                sh "whoami"
                sh "docker build -t flask-new . "
            }
        }

        stage('Test') {
            steps {
                echo 'Running tests...'
            }
        }

        stage('Push'){
            steps{
                withCredentials([usernamePassword(credentialsId:'Dockerhub-cred',usernameVariable:'DOCKERHUB_USER',passwordVariable:'DOCKERHUB_PASS')]){
                    sh '''
                        docker login -u $DOCKERHUB_USER -p $DOCKERHUB_PASS
                        docker tag flask-new:latest $DOCKERHUB_USER/flask-new:latest
                        docker push $DOCKERHUB_USER/flask-new:latest
                        echo "âœ… Image pushed successfully to Docker Hub!"
                    '''
                }
            }
        }

        stage('Deploy') {
            steps {
                echo 'Deploying application...'
                sh "docker-compose up -d "
            }
        }
    }
}
